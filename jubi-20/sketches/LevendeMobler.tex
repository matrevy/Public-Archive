\begin{sketch}
\scene Skuespillere står på scenen, klædt ud som møbler. Kost og stumtjener skjules idet lyset tænder på scenen af et tæppe hængt op i venstre side af scenen.
    (Godfather tema lyder i baggrunden.) 
    \scene En stol sidder på scenen i en  lænestolen.
    
    \says{Stol} Nogen skal dø! ... og det skal ikke være mig.
    
    \scene En kost og dens stumtjener kommer ind.
    
    \says{Kost} Så mødes vi igen, Don Sedia! Jeg kan se du kører den dyre Arne
    Jacobsen-stil.
    
    \says{Stol} Godt set, Ginestra, man skal jo være med på komoden. Jeg vil give dig et tilbud du ikke kan afslå.
    
    \says{Kost} Hvad koster det? \act{sagt som en kost}
    
    \says{Stol} Ikke den slags tilbud.
    
    \says{Kost} Ok, men kan jeg stole på dig? Jeg husker stadig episoden, hvor
    du lå lige lovlig længe og lurede på diva(n)en. Du skulle skamle dig.
    
    \says{Stol} Før du bliver for strid en børste, bør du nok lige feje for egen dør.
    
    \says{Kost} Bare fordi du sidder godt i det, har du ikke ret til at koste rundt med
    mig.
    
    \says{Stol} Din ven siger ikke så meget, hva’?
    
    \says{Kost} Nu synes jeg du skaber dig.
    
    \says{Stol} Reolig nu. Ja, jeg ved du ynder at moppe folk, og du kommer godt ud
    i krogene.
    
    \says{Kost} Det er også temmelig skuffende, at du bliver ved med at hylde mig ud af den.
    
    \says{Stol} Det er ikke mit bord.
    
    \says{Kost} Hvorfor sidder du så ved det.
    
    \says{Stol} Nu skal du ikke borde i det.
    
    \says{Kost} Nej, men så tager jeg da gerne imod tilbudet.
    
    \says{Stol}  Sofa, so good, men nu må der ske noget nyt.
    
    \scene{Stumtjeneren trækker en kniv.}
    
    \says{Stol} Pas på, han har knif!
    
    \scene{Lampen vælter og dræber stumtjeneren.}
    
    \says{Stumtjeneren} Ave Maria, jeg dør \act{sagt som en dør}.
    
    \says{Kost} Der fik han en på lampen.
    
    \says{Stol} Ja, du må hellere skrubbe af.
    
    \scene{Kosten forlader scenen. En klapstol kommer flyvende ind på scenen.}
    
    \says{Stol} klapstol?
    
    \scene{Lænestolen klapper i hænderne. Kosten kommer ind igen}
    
    \scene{Laaang pause}
    
    \says{Stol} Så skal tæppet gå.
    
    \scene{Tæppet der ligger på gulvet (en mand i dynebetræk) går ud.}
\end{sketch}
