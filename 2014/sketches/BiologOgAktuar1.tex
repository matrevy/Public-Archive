\begin{sketch}
\says{V} Biolog og Aktuar.

\scene{Lys på scenen, A sidder ved et bord i fint jakkesæt. Telefonen ringer.}

\says{A} Ja hej, det er aktuar.

\says{B} Hej, det er biolog. Jeg vil gerne anmelde en skade.

\says{A} Ja så gerne, hvilken slags skade er det?

\says{B} Hvad mener du?

\says{A} Altså, er det på bilen, på båden, på huset for eksempel?

\says{B} Ah, det er en husskade.

\says{A} Så gerne, kan du fortælle lidt mere om hvad der er sket?

\says{B} Ja, den fløj direkte ind og stjal mine waders!

\says{A} Hvem gjorde det, siger du?

\says{B} Husskaden selvfølgelig?

\scene{A sukker og lægger på. Lys ned.}
\end{sketch}
