\versetitle{Sangen om humanister}{Hair --- ``Gi' mig sorte drenge''}

\versestart
\settowidth{\versewidth}{Folk med Freud som gud!}
\begin{verse}[\versewidth]
Gi mig humanister\\
Mænd med charmeklud\\
Labre leninister,\\
Halvnøgne nudister,\\
Sorte satanister,\\
Folk med Freud som gud!
\end{verse}
\verseend

\versestart
\settowidth{\versewidth}{Spørg mig ikke hvoffer(, men)}
\begin{verse}[\versewidth]
Flotte filosoffer\\
Er han ikke smuk?\\
Det' det nemme offer,\\
Spørg mig ikke hvoffer(, men)\\
De har det med stoffer, som\\
Vi har det med druk!
\end{verse}
\verseend

\versestart
\settowidth{\versewidth}{Men hans "øjn'" er store!}
\begin{verse}[\versewidth]
Søde sociologer,\\
Der vil studere mig\\
Han bor hos sin moar,\\
Men hans "øjn'" er store!\\
Skal jeg end'lig score,\\
Så skal jeg ha' fransk!
\end{verse}
\verseend

\versestart
\settowidth{\versewidth}{Gi' mig humanister!}
\begin{verse}[\versewidth]
Humanister!\\
Gi' mig humanister!\\
\end{verse}
\verseend
